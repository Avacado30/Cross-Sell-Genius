# Step A: Define the base image to build from.
# We'll use an official, lightweight Python image.
FROM python:3.11-slim

# Step B: Set the working directory inside the container.
# This is where our application files will live.
WORKDIR /app

# Step C: Copy the requirements file into the container.
COPY requirements.txt .

# Step D: Install the Python dependencies.
# This runs 'pip install' inside the container.
RUN pip install --no-cache-dir -r requirements.txt

# Step E: Copy the rest of our application files into the container.
# The '.' means copy everything from the current directory on our computer
# into the WORKDIR ('/app') inside the container.
COPY . .

# Step F: Expose the port the app runs on.
# Our Flask app runs on port 5001.
EXPOSE 5001

# Step G: Define the command to run when the container starts.
# This will start our Flask server.
CMD ["flask", "run", "--host=0.0.0.0", "--port=5001"]
